User lookup
Find user by emailauth.users
SELECT id, email, created_at, last_sign_in_at, deleted_at FROM auth.users WHERE email = 'EMAIL';
All users + signup dateauth.users
SELECT email, created_at, last_sign_in_at FROM auth.users ORDER BY created_at DESC;
Profile + agreement + statusprofiles
SELECT id, email, agreement_signed, country, account_status, created_at FROM profiles WHERE email = 'EMAIL';
Trip data
Trip count + date range for usertrips
SELECT COUNT(*) as trip_count, MIN(trip_date) as oldest, MAX(trip_date) as newest FROM trips WHERE user_id = 'UUID' AND deleted_at IS NULL;
All users + trip countstrips
SELECT u.email, COUNT(t.id) as trips, MIN(t.trip_date) as oldest, MAX(t.trip_date) as newest FROM auth.users u LEFT JOIN trips t ON t.user_id = u.id AND t.deleted_at IS NULL GROUP BY u.email ORDER BY trips DESC;
Last 10 trips for usertrips
SELECT id, from_addr, to_addr, miles, trip_date, created_at FROM trips WHERE user_id = 'UUID' AND deleted_at IS NULL ORDER BY created_at DESC LIMIT 10;
Health check — totals + ID rangehealth
SELECT MIN(id) as min_id, MAX(id) as max_id, COUNT(*) as total, MIN(created_at) as first_trip, MAX(created_at) as last_trip FROM trips WHERE deleted_at IS NULL;
Billing
License status for userlicenses
SELECT status, plan, trial_start, trial_end, billing_path, device_model FROM licenses WHERE user_id = 'UUID';
All active subscriberslicenses
SELECT u.email, l.status, l.plan, l.billing_path, l.trial_end FROM licenses l JOIN auth.users u ON u.id = l.user_id WHERE l.status IN ('active','trialing') ORDER BY l.trial_end ASC;
Trials expiring in 3 daysbilling alert
SELECT u.email, l.trial_end, l.plan FROM licenses l JOIN auth.users u ON u.id = l.user_id WHERE l.status = 'trialing' AND l.trial_end BETWEEN now() AND now() + interval '3 days' ORDER BY l.trial_end ASC;
